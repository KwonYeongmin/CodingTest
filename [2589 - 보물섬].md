#include <iostream>
#include <vector>
#include <queue>
using namespace std;

int N, M;

pair<int, int> BFS(pair<int, int> start, vector<vector<char>>& maps, int& maxDist)
{
    vector<vector<int>> distance(N, vector<int>(M, -1));
    queue<pair<int, int>> q;
    q.push(start);
    distance[start.first][start.second] = 0;
    maxDist = 0;

    int dx[4] = { -1, 1, 0, 0 };
    int dy[4] = { 0, 0, -1, 1 };

    pair<int, int> farthestPoint = start;

    while (false == q.empty()) 
    {
        pair<int, int> cur = q.front();
        q.pop();

        for (int i = 0; i < 4; ++i)
        {
            int nx = cur.first + dx[i];
            int ny = cur.second + dy[i];

            // 범위 계산
            if (nx < 0 || ny < 0 || nx >= N || ny >= M) 
            {
                continue;
            }
            if (maps[nx][ny] == 'W' || distance[nx][ny] != -1) 
            {
                continue;
            }

            distance[nx][ny] = distance[cur.first][cur.second] + 1;
            q.push({ nx, ny });

            if (distance[nx][ny] > maxDist) 
            {
                maxDist = distance[nx][ny];
                farthestPoint = { nx, ny };
            }
        }
    }

    return farthestPoint;
}


int main()
{
    cin >> N >> M;
    vector<vector<char>> maps(N, vector<char>(M, ' '));
  
    for (int i = 0; i < N; ++i) 
    {
        for (int k = 0; k < M; ++k)
        {
            cin >> maps[i][k];
        }
    }

    int maxNum = 0;

    for (int i = 0; i < N; ++i) 
    {
        for (int k = 0; k < M; ++k) 
        {
            if (maps[i][k] == 'L') 
            {  
                // 육지일 경우 BFS 시작
                int dist = 0;
                pair<int, int> farthestPoint = BFS({ i, k }, maps, dist);

                int maxDist = 0;
                BFS(farthestPoint, maps, maxDist);

                maxNum = max(maxNum, maxDist);
            }
        }
    }

    cout << maxNum << '\n';

    return 0;
}
