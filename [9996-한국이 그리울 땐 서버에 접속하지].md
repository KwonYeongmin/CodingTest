```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int N;
    string temp;
    cin >> N;
    cin >> temp;

    vector<string> vec(N, "");
    vector<string> res(N, "NE");

    // aa*bb / a*b -> *앞뒤로 잘라주기
    int idx = temp.find("*");
    string beginStr = temp.substr(0, idx);
    string endStr = temp.substr(idx + 1);

    for (int i = 0; i < N; ++i)
    {
        cin >> vec[i];
        string str = vec[i];

        if (str.length() < beginStr.length() + endStr.length())
        {
            continue;
        }
        if (0 != str.substr(0, beginStr.length()).compare(beginStr) ||
            0 != str.substr(str.length() - endStr.length(), str.length() - 1).compare(endStr))
        {
            continue;
        }

        res[i] = "DA";
    }
    for (int i = 0; i < N; ++i)
    {
        cout << res[i] << '\n';
    }
    return 0;
}
```
