```cpp
#include <iostream>
#include <stack>
#include <vector>

using namespace std;

int main()
{
    stack<int> st;
    int n;
    cin >> n;
    
    vector<char> res;
    
    int num = 1;
    
    for(int i=0; i<n; ++i)
    {
        int x;
        cin >> x;
        
        while(num <= x)
        {
            st.push(num);
            res.push_back('+');
            ++num;
        }
        
        if(st.top() == x)
        {
            st.pop();
            res.push_back('-');
        }
        else
        {
            cout << "NO";
            return 0;
        }
    }
    
    for(char ch : res) 
    {
        cout << ch <<'\n';
    }
    return 0;
}
```
