```cpp
#include <iostream>
using namespace std;

int main()
{
    int A,B,C;
    cin >> A >> B >>C;
    
    int times[101] = {0,};
    int minTime = 0;
    int maxTime = 0;
    
    for(int i=0; i<3; ++i)
    {
        int n1, n2;
        cin >> n1 >> n2;
        
        minTime = min(minTime, n1);
        maxTime = max(maxTime, n2);
        
        for(int j=n1; j<n2; ++j)
        {
            times[j] ++;
        }
    }
    
    int sum = 0;
    for(int i=minTime; i<maxTime; ++i)
    {
        if(times[i] == 1){ sum+=(A*times[i]);}
        else if(times[i] == 2){ sum+=(B*times[i]);}
        else { sum+=(C*times[i]);}
    }
    
    cout << sum;
    
    return 0;
}
```
