```cpp
#include <iostream>
#include <queue>
#include <stack>
using namespace std;

// 문자열에 포함되는 괄호는 소괄호("()") 와 대괄호("[]")로 2종류

int main()
{
    int T;
    cin >> T;
    // (())())
    vector<string> results;

    for (int i = 0; i < T; ++i) 
    {
        string temp;
        cin >> temp;
        stack<char> st;

        for (int k = 0; k < temp.length(); ++k) 
        {
            if (st.empty()) 
            {
                st.push(temp[k]);
            }
            else 
            {
                if (st.top() == '(' && temp[k] == ')')
                {
                    st.pop();
                }
                else
                {
                    st.push(temp[k]);
                }
            }
        }

        if (st.size() == 0) 
        {
            results.push_back("YES");
        }
        else 
        {
            results.push_back("NO");
        }
    }

    for (string str : results) 
    {
        cout << str << '\n';
    }
    return 0;
}
```
