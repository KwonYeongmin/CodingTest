### 비트 연산자
- AND 연산 : &
- OR 연산 : |
- NOT 연산 : ~
- XOR 연산 : ^

</br>

```cpp
#include <string>
#include <vector>
#include <algorithm>
#include <iostream>

using namespace std;
int compareCol;
bool comp(vector<int> vec1, vector<int> vec2)
{
    if(vec1[compareCol] < vec2[compareCol])
    {
        return true;
    }
    else if(vec1[compareCol] == vec2[compareCol])
    {
        return vec1[0] > vec2[0];
    }
    
    return false;
}

int solution(vector<vector<int>> data, int col, int row_begin, int row_end) 
{
    int answer = 0;
    compareCol = col-1;
    // 정렬
    sort(data.begin(), data.end(), comp);
    
    // 해시 값으로 변경
    vector<int> vec;
    for(int i= row_begin-1; i< row_end; ++i)
    {
        int temp =0;
        for(int k=0; k<data[i].size(); ++k)
        {
            temp += data[i][k]%(i+1);
        }
        // XOR 연산 : ^
        answer ^= temp;
    }
    return answer;
}
```
