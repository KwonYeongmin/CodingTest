```cpp
#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main()
{
    int N;
    cin >> N;
    vector<string> vec = vector<string>(N, "");
    vector<int> alphabet = vector<int>(28, 0);

    for (int i = 0; i < N; ++i)
    {
        cin >> vec[i];
    }

    for (int i = 0; i < N; ++i)
    {
        int n = vec[i][0]-97;
        ++alphabet[n];
    }

    string str ="";
    for (size_t i = 0; i < alphabet.size(); ++i) 
    {
        if (5 <= alphabet[i])
        {
            str += (i + 97);
        }
    }

    if (str.empty()) 
    {
        cout << "PREDAJA";
    }
    else 
    {
        cout << str;
    }

    return 0;
}
```
