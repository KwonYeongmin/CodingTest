- 문자열이 끝날 때도 분리되는 문자열 카운팅에 넣어줘야 함 .... !!!!!

```cpp
#include <string>
#include <vector>
#include <iostream>

using namespace std;

int solution(string s) 
{
    int answer = 0;
    int idx = 0;
    
   while (idx < s.length())
    {
        // 변수 셋팅
        char x = s[idx];
        int xCount = 0;
        int otherCount = 0;
        
        for (int i = idx; i < s.length(); ++i)
        {
            // x 
            if (x == s[i]) 
            {
                xCount++;
            }
            // x가 아닌 문자
            else 
            {
                otherCount++;
            }
            
            // 횟수가 같아지는 순간 문자열 분리
            if (xCount == otherCount)
            {
                ++answer;
                idx = i + 1; 
                break;
            }
        }

        // 문자열이 끝날 때 카운팅하기
        if (xCount != otherCount)
        {
            ++answer;
            break;
        }
    }
    
    
    return answer;
}
```
