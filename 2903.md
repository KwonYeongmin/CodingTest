#include <iostream>
#include <cmath>
using namespace std;
/*
    // 0 : 4
    // 1 : f(1) = 3;
    // 2 : f(2) = f(1)*2-1 = 5; f(2)*f(2) = 25
    // 3 : f(3) = f(2)*2-1 = 9;  f(3)*f(3) = 81
    // 4 : f(4) = f(3)*2-1 = 17; f(4)*f(4) = 289
    // 5 : f(5) = f(4)*2-1 = 33; f(4)*f(4) = 196 // 33 * 33
*/

int GetNumber(int num)
{
    if(num==1) return 3;
    return GetNumber(num-1)*2-1;
}

int main()
{
    int N;
    cin >> N;
    int res = pow(GetNumber(N),2);
    cout << res;
    
    return 0;
}
