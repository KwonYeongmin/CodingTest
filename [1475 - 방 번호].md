#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int arr[10] = { 0, };
    int N;
    cin >> N;

    while (N)
    {
        int temp = N % 10;
        ++arr[temp];
        N /= 10;
    }
    
    // int res = (arr[6] + arr[9]) / 2;
    
    // 6과 9의 개수가 홀수일 때에도 정확하게 세트 개수를 계산
    int res = (arr[6] + arr[9] +1) / 2;
    
    arr[6] = 0;  arr[9] = 0;
   
    for (int i = 0; i < 10; i++)
    {
        res = max(arr[i], res);
    }

    cout << res;
    return 0;
}
