### DFS 풀이
```cpp
#include <iostream>
#include <vector>

using namespace std;

int n, m;
int board[501][501];

bool dfs(int x, int y, int& res)
{
    if (x < 0 || y <0 || x > n - 1 || y >m - 1)
    {
        return false;
    }
    if (board[x][y] == 0 )
    {
        return false;
    }
    ++res;
    board[x][y] = 0;
    dfs(x - 1, y,res);
    dfs(x + 1, y,res);
    dfs(x, y - 1,res);
    dfs(x, y + 1,res);
    return true;
}

int main()
{

    cin >> n >> m;

    for (int i = 0; i < n; ++i)
    {
        for (int j = 0; j < m; ++j)
        {
            cin >> board[i][j];
        }
    }

    int cnt = 0;
    int maxNum = 0;
    for (int i = 0; i < n; ++i)
    {
        for (int j = 0; j < m; ++j)
        {
            int res = 0;

            if (dfs(i, j, res)) 
            {
                ++cnt;
            }
            maxNum = max(maxNum, res);
        }
    }

    cout << cnt << '\n' <<maxNum;
    return 0;
}
```
