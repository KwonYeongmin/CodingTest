## DFS 풀이

```cpp
#include <iostream>
#include <queue>
#include <vector>
using namespace std;

// 0은 배추가 심어져 있지 않은 땅이고, 1은 배추가 심어져 있는 땅을 나타낸다.

bool dfs(int x, int y, int M, int N, vector<vector<bool>>& visited, vector<vector<int>> board)
{
    if (x < 0 || y < 0 || x >= M || y >= N) 
    {
        return false;
    }

    if (visited[x][y] || board[x][y] == 0)
    {
        return false;
    }

    visited[x][y] = true;

    dfs(x - 1, y, M, N, visited, board);
    dfs(x + 1, y, M, N, visited, board);
    dfs(x, y - 1, M, N, visited, board);
    dfs(x, y + 1, M, N, visited, board);
    
    return true;
}

int main()
{
    int T;
    int M, N;
    int K;
    cin >> T;

    vector<int> results;

    for (int i = 0; i < T; ++i)
    {
        cin >> M >> N >> K;

        vector<vector<bool>> visited(M, vector<bool>(N, false));
        vector<vector<int>> board(M, vector<int>(N,0));

        for (int j = 0; j < K; ++j)
        {
            int x, y;
            cin >> x >> y;
            board[x][y] = 1;
        }

        int res = 0;
        for (int j = 0; j < M; ++j) 
        {
            for (int k = 0; k < N; ++k) 
            {
                if (dfs(j,k, M, N, visited, board))
                {
                    ++res;
                }
            }
        }

        results.push_back(res);
    }

    for (int res : results)
    {
        cout << res << '\n';
    }

    return 0;
}
```
