### 풀이
```cpp
#include <iostream>
#include <queue>
using namespace std;

struct Node 
{
    int x, y;
    Node() :x(0), y(0) {  }
    Node(int _x, int _y) :x(_x), y(_y) {  }
};

int R, C;
char board[1002][1002];

int fire_time[1002][1002];
int jh_time[1002][1002];

queue<Node> fire_q;
queue<Node> jh_q;

// 방향 벡터
int dx[4] = { 0,0,-1,1 };
int dy[4] = { -1,1 ,0,0};

int bfs_jh() 
{
    while (false == jh_q.empty()) 
    {
        int x = jh_q.front().x;
        int y = jh_q.front().y;
        jh_q.pop();

        // 탈출 조건
        if (x == 0 || y == 0 || x == R - 1 || y == C-1) 
        {
            return jh_time[x][y];
        }
        for (int i = 0; i < 4; ++i) 
        {
            int nx = dx[i] + x;
            int ny = dy[i] + y;

            if (nx < 0 || ny < 0 || nx >= R || ny >= C)
            {
                continue;
            }

            // 방문한 칸이거나 벽이라면
            if (jh_time[nx][ny] > 0 || board[nx][ny] == '#') 
            {
                continue;
            }

            // 불이 있는 칸이거나, 불이 먼저 왔다간 곳이라면
            if (fire_time[nx][ny] != 0 && fire_time[nx][ny] <= jh_time[x][y] + 1) 
            {
                continue;
            }

            jh_q.push(Node(nx,ny));
            jh_time[nx][ny] = jh_time[x][y] + 1;
        }
    }
    return -1;
}
void bfs_fire() 
{
    while (false == fire_q.empty())
    {
        int x = fire_q.front().x;
        int y = fire_q.front().y;
        fire_q.pop();

        for (int i = 0; i < 4; ++i)
        {
            int nx = dx[i] + x;
            int ny = dy[i] + y;

            if (nx < 0 || ny < 0 || nx >= R || ny >= C)
            {
                continue;
            }

            // 방문한 칸이거나 벽이라면
            if (fire_time[nx][ny] > 0 || board[nx][ny] == '#')
            {
                continue;
            }

            fire_q.push(Node(nx, ny));
            fire_time[nx][ny] = fire_time[x][y] + 1;
        }
    }
}


int main()
{
    cin >> R >> C;
    for (int i = 0; i < R; ++i) 
    {
        for (int j = 0; j < C; ++j)
        {
            cin >> board[i][j];

            if (board[i][j] == 'J') 
            {
                jh_q.push(Node(i, j));
                jh_time[i][j] = 1;
            }
            if (board[i][j] == 'F')
            {
                fire_q.push(Node(i, j));
                fire_time[i][j] = 1;
            }
        }
    }

    // 불 이동
    bfs_fire();

    // 지훈 이동
    int res = bfs_jh();

    if (res == -1)
    {
        cout << "INPOSSIBLE";
        return 0;
    }
    cout << res;
    return 0;
}
```
