```cpp
#include <iostream>
#include <queue>

using namespace std;

// 지훈이의 위치와 불이 붙은 위치를 감안해서 지훈이가 불에 타기전에 탈출할 수 있는지의 여부, 
// 얼마나 빨리 탈출할 수 있는지를 결정

struct Node
{
    int x, y;
    Node() : x(0), y(0) {  }
    Node(int _x, int _y) : x(_x), y(_y) {  }
};

int R, C;
char board[1002][1002];
int jh_time[1002][1002];
int fire_time[1002][1002];

queue<Node> q_jh;
queue<Node> q_fire;

int dx[4] = { 0,0,-1,1 };
int dy[4] = { -1,1,0,0 };

void bfs_fire()
{
    while (false == q_fire.empty())
    {
       int x = q_fire.front().x;
       int y = q_fire.front().y;
        q_fire.pop();

        for (int i = 0; i < 4; ++i)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];

            // 범위를 벗어나는 경우
            if (nx < 0 || ny < 0 || nx >= R || ny >= C)
            {
                continue;
            }

            // 벽에 도달하거나 이미 방문한 경우
            if (board[nx][ny] == '#' || fire_time[nx][ny] >= 0)
            {
                continue;
            }

            fire_time[nx][ny] = fire_time[x][y] + 1;
            q_fire.push(Node(nx, ny));
        }
    }
}

int bfs_jh()
{
    while (false == q_jh.empty())
    {
        int x = q_jh.front().x;
        int y = q_jh.front().y;
        q_jh.pop();

        // 탈출 조건 : 가장 자리에 도착하면 탈출
        if (x == 0 || y == 0 || x == R - 1 || y == C - 1) 
        {
            return jh_time[x][y] + 1;
        }

        for (int i = 0; i < 4; ++i) 
        {
            int nx = x + dx[i];
            int ny = y + dy[i];

            // 범위를 벗어나는 경우
            if (nx < 0 || ny < 0 || nx >= R || ny >= C) 
            {
                continue;
            }

            // 벽에 도달하거나 이미 방문한 경우
            if (board[nx][ny] == '#' || jh_time[nx][ny] >= 0)
            {
                continue;
            }

            jh_time[nx][ny] = jh_time[x][y] + 1;

            // 지훈이가 이동하려는 칸에 불이 먼저 퍼지는 경우
                // 이동하려는 칸에 불이 퍼졌거나, 지훈이가 불이 퍼진 뒤에 도착한 경우
            if (fire_time[nx][ny] != -1 && (fire_time[nx][ny] <= jh_time[nx][ny]))
            {
                continue;
            }

            q_jh.push(Node(nx, ny));
        }
    }

    // 탈출할 수 없을 경우
    return -1;
}

int main()
{
    cin >> R >> C;

    for (int i = 0; i < R; ++i)
    {
        for (int j = 0; j < C; ++j)
        {
            cin >> board[i][j];
            jh_time[i][j] = -1;
            fire_time[i][j] = -1;

            if (board[i][j] == 'J') 
            {
                q_jh.push(Node(i,j));
                jh_time[i][j] = 0;
            }

            if (board[i][j] == 'F')
            {
                q_fire.push(Node(i, j));
                fire_time[i][j] = 0;
            }
        }
    }

    bfs_fire();
    int res =bfs_jh();
    
    if (res == -1)
    {
        cout << "IMPOSSIBLE"; 
    }
    else 
    {
        cout << res;
    }

    return 0;
}
```
