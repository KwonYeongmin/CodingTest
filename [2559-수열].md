
 ### 구간 합 풀이
'''cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int K, N;
    cin >> K >> N;
    vector<int> arr(K, 0);

    for (int i = 0; i < K; ++i)
    {
        cin >> arr[i];
    }
    
    int maxNum = -10000001;

    for (int i = 0; i <= K-N; i++)
    {
        int sum = 0;
        
        for (int j = i; j < i + N; ++j) 
        {
            sum += arr[j];
        }
        maxNum = max(maxNum, sum);
    }
    cout << maxNum;
    return 0;
}

'''
