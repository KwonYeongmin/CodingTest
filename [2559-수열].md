### 합계 누적 풀이
```cpp
#include <iostream>

using namespace std;

int main()
{
    int N,K;
    cin >> N >> K;
    
    int res = -10000001;
    int pSum[100001] ={0,};
    
    for(int i=1; i<N+1; ++i )
    {
        int temp;
        cin >> temp;
        pSum[i] = pSum[i-1] + temp;
    }
    
    for(int i = K; i <= N; ++i)
    {
        res = max(res, pSum[i] - pSum[i-K]);
    }
    cout << res;
    return 0;
}
```

 ### 풀이2
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int K, N;
    cin >> K >> N;
    vector<int> arr(K, 0);

    for (int i = 0; i < K; ++i)
    {
        cin >> arr[i];
    }
    
    int maxNum = -10000001;

    for (int i = 0; i <= K-N; i++)
    {
        int sum = 0;
        
        for (int j = i; j < i + N; ++j) 
        {
            sum += arr[j];
        }
        maxNum = max(maxNum, sum);
    }
    cout << maxNum;
    return 0;
}
```
