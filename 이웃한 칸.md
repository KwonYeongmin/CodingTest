```cpp
#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<string>> board, int h, int w) 
{
    int answer = 0;
    
    int dx[4] = {0,0,-1,1};
    int dy[4] = {-1,1,0,0};
    
    string compareColor = board[h][w];
    
    for(int i=0; i<4; ++i)
    {
        int x = dx[i] + h;
        int y = dy[i] + w;
        
        if(x <0 || y < 0 || x > board.size()-1 || y > board[0].size()-1)
        {
            continue;
        }
        
        if(compareColor.compare(board[x][y]) == 0)
        {
            ++ answer;
        }
    }
    
    
    return answer;
}
```
