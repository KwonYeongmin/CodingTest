```cpp
#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    vector<string> res = vector<string>(15,"");
    int maxIndex = 0;
    for (int i = 0; i < 5; ++i) 
    {
        string str;
        cin >> str;
        if (maxIndex < str.length()) 
        {
            maxIndex = str.length();
        }
        for (int k = 0; k < str.length(); ++k)
        {
            res[k] += str[k];
        }
    }
    for (int i = 0; i < maxIndex; ++i) 
    {
        cout << res[i];
    }
    return 0;
}
```
