## 문제
https://www.acmicpc.net/problem/10816

### 문제 풀이
- 풀이법 1 -> map을 이용하여 풀기
- 풀이법 2 -> 배열을 사용해서 풀기
- 풀이법 2 -> 이진탐색 사용해서 풀기
- 풀이법 3 -> STL 알고리즘 사용해서 풀기

## 풀이
### map을 사용해서 풀기
- 시간 : 712ms
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <map>

using namespace std;


int main()
{
    ios_base :: sync_with_stdio(0); 
    cin.tie(0); 
    cout.tie(0);
    
	int N, M;
	vector<int> arr;
	vector<int> cards;
	map<int, int> mp;

	// 상근이가 가지고 있는 카드
	cin >> N; 
	for (int i = 0; i < N; ++i)
	{
		int t;
		cin >> t;
		arr.push_back(t);
	}

	cin >> M;
	for (int i = 0; i < M; ++i)
	{
		int t;
		cin >> t;
		mp[t] = 0;
		cards.push_back(t);
	}

	for (int i = 0; i < N; ++i)
	{
		if (mp.find(arr[i]) != mp.end())
		{
			mp[arr[i]]++;
		}
	}
	
	for (int i = 0; i < M; ++i)
	{
		cout << mp[cards[i]] << ' ';
	}

	return 0;
}
```

### 배열 사용
- 시간 : 272ms
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <map>

using namespace std;

int cards[20000002];
int arr[20000002];

int main()
{
	int N, M;

	ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);

	// 상근이가 가지고 있는 카드
	cin >> N; 
	for (int i = 0; i < N; ++i)
	{
		int t;
		cin >> t;
		cards[t]++;
	}

	cin >> M;
	for (int i = 0; i < M; ++i)
	{
		cin >> arr[i];
	}

	for (int i = 0; i < M; ++i)
	{
		cout << cards[arr[i]] << ' ';
	}

	return 0;
}
```
