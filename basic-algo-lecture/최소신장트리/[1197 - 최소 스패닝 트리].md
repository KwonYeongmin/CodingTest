## 문제
https://www.acmicpc.net/problem/1197

## 문제 풀
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int V,E;

struct Edge 
{
    int u,v, cost;
    Edge(int _u, int _v, int _cost) : u(_u),v(_v), cost(_cost) {}
};

int parent[10002];
vector<Edge> edges;

int GetParent(int x) 
{
    if (parent[x] == x) 
    {
        return x;
    }
    return parent[x] = GetParent(parent[x]);
}

void UnionParent(int a, int b)
{
    a = GetParent(a);
    b = GetParent(b);
    if (a < b) 
    {
        parent[b] = a;
    }
    else 
    {
        parent[a] = b;
    }
}

bool FindParent(int a, int b) 
{
    a = GetParent(a);
    b = GetParent(b);
    if (a == b) 
    {
        return true;
    }
    return false;
}

bool comp(Edge a, Edge b) 
{
    return a.cost < b.cost;
}

int main()
{
    cin.tie(NULL);
    cout.tie(NULL);
    ios_base::sync_with_stdio(false);

    cin >> V >> E;

    for (int i = 1; i <= V; ++i) 
    {
        parent[i] = i;
    }
    
    for (int i = 0; i < E; ++i) 
    {
        int a, b, c;
        cin >> a >> b >> c;
        edges.push_back(Edge(a,b,c));
    }

    sort(edges.begin(), edges.end(), comp);

    int result = 0;
    int cnt = 0;
    for (Edge edge : edges)
    {
        int a = edge.u;
        int b = edge.v;
        if (FindParent(a, b))
        {
            continue;
        }
        UnionParent(a, b);
        result += edge.cost;
        ++cnt;
        if (cnt == V - 1)
        {
            break;
        }
    }

    cout << result;

    return 0;
}

```
