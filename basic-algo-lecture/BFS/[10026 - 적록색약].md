## BFS 
- 만약 N의 범위가 100 이상으로 크게 증가한다면 DFS보다는 BFS를 사용하는 게 더 적절하다.
### BFS 풀기
```cpp
#include <iostream>
#include <queue>
#include <string>

using namespace std;

struct Node 
{
    int x, y;
    Node() : x(0), y(0) {}
    Node(int _x, int _y) : x(_x), y(_y) {}
};

int N;
int board[102][102][2];
bool visited[102][102][2];
queue<Node> qR;
queue<Node> qG;
queue<Node> qB;
queue<Node> qRG;

int dx[4] = { 0,0,-1,1 };
int dy[4] = { -1,1,0,0 };

int red_normal = 0;
int green_normal = 0;
int blue = 0;
int redGreen = 0;

void bfs(int x, int y, int color, int flag)
{
    queue<Node> q;
    q.push(Node(x, y));
    visited[x][y][flag] = true;

    while (!q.empty()) 
    {
        int x = q.front().x;
        int y = q.front().y;
        q.pop();

        for (int i = 0; i < 4; ++i)
        {
            int nx = dx[i] + x;
            int ny = dy[i] + y;

            if (nx < 0 || ny < 0 || nx >= N || ny >= N) 
            {
                continue;
            }

            if (board[nx][ny][flag] != color || visited[nx][ny][flag])
            {
                continue;
            }

            visited[nx][ny][flag] = true;
            q.push(Node(nx, ny));
        }
    }
}

int main()
{
    cin >> N;
    for (int i = 0; i < N; ++i) 
    {
        for (int j = 0; j < N; ++j)
        {
            char temp;
            cin >> temp;
            if (temp == 'B') 
            {
                board[i][j][0] = 0;
                board[i][j][1] = 0;
                qB.push(Node(i, j)); 
            }
            else if (temp == 'R') 
            {
                board[i][j][0] = 1;
                board[i][j][1] = 1;
                qR.push(Node(i, j));
                qRG.push(Node(i,j));
            }
            else 
            {
                board[i][j][0] = 2;
                board[i][j][1] = 1;
                qG.push(Node(i, j));
                qRG.push(Node(i, j));
            }
        }
    }
    int res[2] = { 0, };
    for (int i = 0; i < N; ++i)
    {
        for (int j = 0; j < N; ++j)
        {
            for (int k = 0; k < 2; ++k) 
            {
                if (visited[i][j][k] == false)
                {
                    bfs(i, j, board[i][j][k], k);
                    ++res[k];
                }
            }
        }
    }

    for (int k = 0; k < 2; ++k)
    {
        cout << res[k] << ' ';
    }

    return 0;
}

```

## DFS
### DFS 풀이
```cpp
#include <iostream>
#include <string>

using namespace std;

// 적록색약은 빨간색과 초록색의 차이를 거의 느끼지 못한다. 
//  파란색  0 빨간색 1 초록색 2
//  파란색  0 빨간색 1 초록색 1
int board[102][102][2];
bool visited[102][102][2];
int N;


bool dfs(int x, int y, int color, int flag)
{
    if (x < 0 || y < 0 || x >= N || y >= N) 
    {
        return false;
    }

    if (visited[x][y][flag] || board[x][y][flag] != color)
    {
        return false;
    }
    visited[x][y][flag] = true;
    dfs(x, y - 1, color,flag);
    dfs(x, y + 1, color, flag);
    dfs(x - 1, y, color, flag);
    dfs(x + 1, y, color, flag);

    return true;
}

int main() 
{
    cin >> N;

    for (int i = 0; i < N; ++i) 
    {
        for (int j = 0; j < N; ++j)
        {
            char temp;
            cin >> temp;

            if (temp == 'B') 
            {
                board[i][j][0] = 0;
                board[i][j][1] = 0;
            }
            else if (temp == 'R')
            {
                board[i][j][0] = 1;
                board[i][j][1] = 1;
            }
            else
            {
                board[i][j][0] = 2;
                board[i][j][1] = 1;
            }
        }
    }
    int res[2] = { 0 , };
     
    for (int i = 0; i < N; ++i)
    {
        for (int j = 0; j < N; ++j)
        { 
            for (int k = 0; k < 2; ++k) 
            {
                if (dfs( i, j, board[i][j][k], k))
                {
                    ++res[k];
                }
            }
        }
    }

    for (int k = 0; k < 2; ++k)
    {
        cout << res[k] << ' ';
    }
    return 0;
}
```
