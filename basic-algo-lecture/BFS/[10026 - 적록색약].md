## DFS 풀이
```cpp
#include <iostream>
#include <string>

using namespace std;

// 적록색약은 빨간색과 초록색의 차이를 거의 느끼지 못한다. 
//  파란색  0 빨간색 1 초록색 2
//  파란색  0 빨간색 1 초록색 1
int board[102][102][2];
bool visited[102][102][2];
int N;


bool dfs(int x, int y, int color, int flag)
{
    if (x < 0 || y < 0 || x >= N || y >= N) 
    {
        return false;
    }

    if (visited[x][y][flag] || board[x][y][flag] != color)
    {
        return false;
    }
    visited[x][y][flag] = true;
    dfs(x, y - 1, color,flag);
    dfs(x, y + 1, color, flag);
    dfs(x - 1, y, color, flag);
    dfs(x + 1, y, color, flag);

    return true;
}

int main() 
{
    cin >> N;

    for (int i = 0; i < N; ++i) 
    {
        for (int j = 0; j < N; ++j)
        {
            char temp;
            cin >> temp;

            if (temp == 'B') 
            {
                board[i][j][0] = 0;
                board[i][j][1] = 0;
            }
            else if (temp == 'R')
            {
                board[i][j][0] = 1;
                board[i][j][1] = 1;
            }
            else
            {
                board[i][j][0] = 2;
                board[i][j][1] = 1;
            }
        }
    }
    int res[2] = { 0 , };
     
    for (int i = 0; i < N; ++i)
    {
        for (int j = 0; j < N; ++j)
        { 
            for (int k = 0; k < 2; ++k) 
            {
                if (dfs( i, j, board[i][j][k], k))
                {
                    ++res[k];
                }
            }
        }
    }

    for (int k = 0; k < 2; ++k)
    {
        cout << res[k] << ' ';
    }
    return 0;
}
```
