# 진법 변환
## 역순 string에 저장한 다음 출력하기
```cpp
#include <iostream>
#include <string>
using namespace std;

int main()
{
    int N, B;
    cin >> N >> B;
    string str = "";
    while (N > 0)
    {
        int num = (N % B);

        // 10이상은 알파벳 대문자로 대체해야한다.
        if (num >= 10)
        {
            str += ((num - 10) + 'A');
        }
        // 10미만의 숫자는 문자로 바꿔야 한다.
        else
        {
            str += (num + '0');
        }
        N = N/B;
    }
    string res = "";
    // 문자열을 뒤집어 준다.
   for (int i = 0; i < str.length(); ++i) 
   {
       res += str[str.length() - 1 - i];
   }

   cout << res;
   return 0;
}
```
## 역순으로 바로 출력하기


```cpp
int main()
{
    int N, B;
    cin >> N >> B;
    string res = "";
    while (N > 0) 
    {
        int num = N % B;
        if (num >= 10) 
        {
            res += (num+'A'-10);
        }
        else 
        {
            res += (num + '0');
        }
        N /= B;
    }

    for(int i=res.length()-1; i>=0;--i)
    {
        cout << res[i];
    }
   
    return 0;
}
```
