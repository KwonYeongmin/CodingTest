int main()
{
    deque<int> dq;
    int N, M;
    cin >> N >> M;
    for (int i = 1; i <= N; ++i)
    {
        dq.push_back(i);
    }

    int res = 0;

    for(int i=0; i<M; ++i)
    {
        int target;
        cin >> target;

        // 목표 숫자가 있는 위치를 찾음
        int idx = find(dq.begin(), dq.end(), target) - dq.begin(); // 0-based index로 변환.

        // 목표 숫자가 덱의 맨 앞에 올 때까지 반복
        while (dq.front() != target)
        {
            // 뒤에서 더 가까울 경우
            if (idx < dq.size() - idx) 
            {
                // 왼쪽으로 한 칸씩 회전
                dq.push_back(dq.front());
                dq.pop_front();
            }
            // 앞에서 더 가깝다면
            else 
            {
                //오른쪽으로 한 칸씩 회전
                dq.push_front(dq.back());
                dq.pop_back();
            }

            // 연산 횟수 증가
            ++res; 
        }

        // 목표 숫자를 뽑았으므로 front에서 제거
        dq.pop_front();
    }
    std::cout << res;
    
    return 0;
}
